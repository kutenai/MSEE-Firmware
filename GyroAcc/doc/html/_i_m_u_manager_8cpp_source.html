00198 int IMUManager::Loop()
00199 {
00200     switch(_State) {
00201     case sIdle:
00202         break;
00203     case sDataWait:
00204         if (DataReady()) {
00205             ResetDataReadyTO();
00206             _State = sDataReady;
00207         } else if (DataReadyTimeout()) {
00208             ResetDataReadyTO();
00209             _State = sDataTimeout;
00210         }
00211         break;
00212     case sDataReady:
00213         if (_nStreamCount) {
00214             SendPacket(false);
00215             --_nStreamCount;
00216         } else {
00217             DiscardData();
00218         }
00219         _State = sDataWait;
00220         break;
00221     case sDataTimeout:
00222         if (_nStreamCount) {
00223             SendPacket(true);
00224             --_nStreamCount;
00225         }
00226         _State = sDataWait;
00227         break;
00228     }

00230     return 0;
